## 数组赋值
## 单个元素赋值
# a[0]=zero
# a[1]=one
# a[2]=two
 
## 定义整个数组
# b=(1 2 3)
 
 
## 查看数组的值
## 查看单个元素的值
# echo ${a[0]}
zero
 
## 查看整个数组的值
# echo ${b[*]}
1 2 3
# echo ${a[@]}
zero one two
 
## 查看数组长度
## 查看单个元素的内容长度
# echo ${a[0]}
zero
# echo ${#a[0]}
4
 
## 查看整个数组的长度
# echo ${b[@]}
1 2 3
# echo ${#b[@]}
3
